ax2 + bx + c = 0 // ax2 - x в квадрате

D = b2 - 4ac

x1 = (-b + sqrt(D)) / 2a
x2 = (-b - sqrt(D)) / 2a

# Матрицы

## Уножение матриц
Матрицы можно перемножать, если кол-во столбцов первой матрицы равно кол-ву строк во второй матрице.
A = [
    [1, 2],
    [3, 4]
]

B = [
    [5, 6, 7],
    [8, 9, 10]
]

A x B = [
     [1*5+2*8, 1*6+2*9, 1*7+2*10],
     [3*5+4*8, 3*6+4*9, 3*7+4*10],
]
